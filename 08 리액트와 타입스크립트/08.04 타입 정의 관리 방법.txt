같은 타입 정의를 여러 위치에서 사용하는 경우 하나의 타입을 여러 컴포넌트에서 재사용하면 된다.

export interface Comment {
  id: number;
  name: string;
  email: string;
  body?: string; // 사용하는 측에서 값이 props로 넘어오지 않아도 undefined 값으로 받음
}


import type { Comment } from "../../interfaces/comment";
//import 뒤에 type이 나오는데 이것은 명시적 타입 정의만 import하기 위한 구문이다.
//type을 붙이지 않아도 되긴 하지만, 타입 정의 import시에는 사용하는 걸 권장한다.

export const ListItem = (props: Comment) => {
  const { id, name, email, body = "default" } = props;
  return (
    <p>
      {id}: {name}({email}) / body: {body}
    </p>
  );
};
